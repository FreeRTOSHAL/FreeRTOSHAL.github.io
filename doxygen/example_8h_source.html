155  /*
156  * Define Global Array in local space

158  */
159  HAL_DEFINE_GLOBAL_ARRAY(example);
160  /*
161  * Get Dev from array if NULL dev not exists

163  struct example_generic *p = (struct example_generic *) HAL_GET_DEV(example, index);
164  if (p == NULL) {
165  return NULL;

168 }
