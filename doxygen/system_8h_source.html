163 #define swap32(d) ({ \
164  union {uint32_t ret; uint8_t ret8[4];} tmp;\
165  tmp.ret = (d); \
166  /* swap 0 -> 3 and 1 -> 2 -> 0123 -> 3210 */ \
167  /* use XOR Swap */ \
168  tmp.ret8[0] ^= tmp.ret8[3]; \
169  tmp.ret8[3] ^= tmp.ret8[0]; \
170  tmp.ret8[0] ^= tmp.ret8[3]; \
171  tmp.ret8[1] ^= tmp.ret8[2]; \
172  tmp.ret8[2] ^= tmp.ret8[1]; \
173  tmp.ret8[1] ^= tmp.ret8[2]; \
174  tmp.ret; \
175 })
