86 inline struct vdev *vdev_init(uint32_t index, struct virtio *virtio, uint8_t size, struct vring_config *config) {
87  HAL_DEFINE_GLOBAL_ARRAY(vdev);
88  struct vdev_generic *gen = HAL_GET_DEV(vdev, index);
89  if (gen == NULL) {
90  return NULL;
91  }
92  return gen->ops->vdev_init(index, virtio, size, config);
93 }
